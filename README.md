# customer_shopping_behaviour

SELECT *
FROM customer_shopping;

UPDATE customer_shopping
SET festival_sale = 'Standard Day'
WHERE festival_sale IS NULL;

COMMIT;

--Orders placed gender wise
SELECT DISTINCT(GENDER), COUNT(*)
FROM CUSTOMER_SHOPPING
GROUP BY GENDER;

--SALES YEARS
SELECT MIN(PURCHASE_DATE) AS EARLY_PURCHASE,
MAX(PURCHASE_DATE) AS LATE_PURCHASE
FROM CUSTOMER_SHOPPING
;


--yearly revenue and other findings
SELECT 
    TO_CHAR(PURCHASE_DATE,'YYYY') AS YEAR,
    COUNT(*) AS TOTAL_ORDERS,
    SUM(PURCHASE_AMOUNT) AS TOTAL_REVENUE,
    ROUND(AVG(PURCHASE_AMOUNT),2) AS AVG_ORDER_VALUE
FROM CUSTOMER_SHOPPING
GROUP BY TO_CHAR(PURCHASE_DATE,'YYYY')
ORDER BY YEAR;
--store wise revenue each year
SELECT 
    TO_CHAR(PURCHASE_DATE,'YYYY') AS YEAR,
    ONLINE_STORE,
    SUM(PURCHASE_AMOUNT) AS TOTAL_REVENUE
FROM CUSTOMER_SHOPPING
GROUP BY ONLINE_STORE ,TO_CHAR(PURCHASE_DATE,'YYYY')
ORDER BY YEAR DESC;

--stores with revenue
SELECT ONLINE_OFFLINE, SUM(PURCHASE_AMOUNT) AS REVENUE
FROM CUSTOMER_SHOPPING
GROUP BY ONLINE_OFFLINE;

--online_store revenue category wise
SELECT ONLINE_STORE, CATEGORY , SUM(PURCHASE_AMOUNT) as total_revenue_per_store
FROM CUSTOMER_SHOPPING
WHERE RETURN_STATUS = 'Not Returned' 
GROUP BY category,ONLINE_STORE
order by CATEGORY;
